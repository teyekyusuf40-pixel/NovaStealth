import customtkinter as ctk import os import sys import ctypes # Siyah ekranı ve izin sorununu kökten çözen başlangıç def run_as_admin(): if ctypes.windll.shell32.IsUserAnAdmin(): return True else: # CMD penceresi olmadan (0 parametresiyle) yönetici izni ister ctypes.windll.shell32.ShellExecuteW(None, "runas", sys.executable, " ".join(sys.argv), None, 0) sys.exit() class NovaStealthElite(ctk.CTk): def __init__(self): super().__init__() self.title("NovaOS - Ghost Protocol Elite v4.0") self.geometry("550x700") self.resizable(False, False) ctk.set_appearance_mode("dark") # Ana Çerçeve self.container = ctk.CTkFrame(self, corner_radius=20, border_width=2, border_color="#00FF00") self.container.pack(pady=20, padx=20, fill="both", expand=True) self.header = ctk.CTkLabel(self.container, text="🛡️ NOVA STEALTH PRO", font=("Orbitron", 28, "bold"), text_color="#00FF00") self.header.pack(pady=25) # 10 Ülke ve Proxy Listesi self.proxies = { "Almanya (Hızlı)": "88.99.10.251:1080", "Hollanda": "188.166.162.33:8080", "ABD - New York": "144.202.112.152:80", "İngiltere": "178.62.193.19:3128", "Fransa": "51.15.37.154:3128", "Japonya": "103.208.200.12:80", "Singapur": "139.59.105.151:8080", "Kanada": "158.69.53.111:3128", "İtalya": "185.241.208.204:80", "Brezilya": "18.228.198.150:80" } # Ülke Seçimi self.c_label = ctk.CTkLabel(self.container, text="LOKASYON SEÇİMİ", font=("Arial", 12, "bold")) self.c_label.pack(pady=(10, 0)) self.country_menu = ctk.CTkOptionMenu(self.container, values=list(self.proxies.keys()), fg_color="#1f538d", width=250) self.country_menu.pack(pady=10) # IP Gizleme Butonu self.ip_btn = ctk.CTkButton(self.container, text="IP GİZLEMEYİ BAŞLAT", fg_color="#1b4d1b", hover_color="#006400", command=self.activate_proxy, height=45) self.ip_btn.pack(pady=15, padx=60, fill="x") # Güvenlik Modülleri self.firewall_active = False self.f_btn = ctk.CTkButton(self.container, text="DIŞARIYI BLOKE ET [KAPALI]", fg_color="#333", command=self.toggle_firewall) self.f_btn.pack(pady=10, padx=60, fill="x") self.dns_active = False self.d_btn = ctk.CTkButton(self.container, text="GİZLİ DNS [KAPALI]", fg_color="#333", command=self.toggle_dns) self.d_btn.pack(pady=10, padx=60, fill="x") # --- KRİTİK: ACİL DURUM SIFIRLAMA --- self.reset_btn = ctk.CTkButton(self.container, text="SİSTEMİ TAM ONAR (GAMING MODE)", fg_color="#8B0000", hover_color="#FF0000", command=self.full_reset, height=50, font=("Arial", 14, "bold")) self.reset_btn.pack(pady=30, padx=60, fill="x") self.status = ctk.CTkLabel(self.container, text="DURUM: SİSTEM NORMAL", text_color="gray") self.status.pack(side="bottom", pady=15) # Kapanış Protokolü self.protocol("WM_DELETE_WINDOW", self.close_safe) def activate_proxy(self): choice = self.country_menu.get() p_addr = self.proxies[choice] os.system(f'reg add "HKCU\Software\Microsoft\Windows\CurrentVersion\Internet Settings" /v ProxyEnable /t REG_DWORD /d 1 /f') os.system(f'reg add "HKCU\Software\Microsoft\Windows\CurrentVersion\Internet Settings" /v ProxyServer /t REG_SZ /d "{p_addr}" /f') self.status.configure(text=f"AKTİF: {choice}", text_color="#00FF00") self.ip_btn.configure(text="IP GİZLENDİ ✔", fg_color="#00FF00", text_color="black") def toggle_firewall(self): self.firewall_active = not self.firewall_active if self.firewall_active: os.system('netsh advfirewall firewall add rule name="NovaGuard" dir=out action=block') self.f_btn.configure(text="DIŞARI BLOKE EDİLDİ ✔", fg_color="#1b4d1b") else: os.system('netsh advfirewall firewall delete rule name="NovaGuard"') self.f_btn.configure(text="DIŞARIYI BLOKE ET [KAPALI]", fg_color="#333") def toggle_dns(self): self.dns_active = not self.dns_active if self.dns_active: os.system('netsh interface ip set dns "Wi-Fi" static 1.1.1.1') self.d_btn.configure(text="DNS ŞİFRELENDİ ✔", fg_color="#1b4d1b") else: os.system('netsh interface ip set dns "Wi-Fi" dhcp') self.d_btn.configure(text="GİZLİ DNS [KAPALI]", fg_color="#333") def full_reset(self): # Her şeyi temizle os.system('reg add "HKCU\Software\Microsoft\Windows\CurrentVersion\Internet Settings" /v ProxyEnable /t REG_DWORD /d 0 /f') os.system('netsh advfirewall firewall delete rule name="NovaGuard"') os.system('netsh interface ip set dns "Wi-Fi" dhcp') os.system('netsh int ip reset') # UI Güncelle self.status.configure(text="DURUM: SİSTEM ONARILDI", text_color="white") self.ip_btn.configure(text="IP GİZLEMEYİ BAŞLAT", fg_color="#1b4d1b", text_color="white") self.f_btn.configure(text="DIŞARIYI BLOKE ET [KAPALI]", fg_color="#333") self.d_btn.configure(text="GİZLİ DNS [KAPALI]", fg_color="#333") self.firewall_active = False self.dns_active = False def close_safe(self): # Kapatırken her şeyi sıfırla ki internet gitmesin self.full_reset() self.destroy() if __name__ == "__main__": if run_as_admin(): app = NovaStealthElite() app.mainloop()
